1. Run Rabbit MQ
docker run -d --hostname my-rabbit --name ecomm-rabbit -p 15672:15672 -p 5672:5672 rabbitmq:3-management

2. Run Redis
docker pull redis

docker stop my-redis
docker rm my-redis
